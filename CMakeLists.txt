cmake_minimum_required(VERSION 3.11)
project(paraCL)

find_package(BISON REQUIRED)
find_package(FLEX REQUIRED)


set(CMAKE_CXX_STANDARD_REQUIRED 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_CXX_STANDARD 20)
add_subdirectory(tests)

set (CMAKE_CURRENT_SOURCE_DIR ${CMAKE_SOURCE_DIR}/src)
include_directories(${CMAKE_SOURCE_DIR}/include)


flex_target(scanner
  ${CMAKE_CURRENT_SOURCE_DIR}/lex_rules.l
  ${CMAKE_CURRENT_BINARY_DIR}/lexer_autogenerated.cc
)

bison_target(parser
  ${CMAKE_CURRENT_SOURCE_DIR}/pcl_grammar.y
  ${CMAKE_CURRENT_BINARY_DIR}/parser.cc
  COMPILE_FLAGS "--defines=${CMAKE_CURRENT_BINARY_DIR}/pcl_grammar.tab.hh"
)

add_flex_bison_dependency(scanner parser)

add_executable(${PROJECT_NAME}
  ${CMAKE_CURRENT_SOURCE_DIR}/driver.cpp
  ${BISON_parser_OUTPUTS}
  ${FLEX_scanner_OUTPUTS}
)

target_compile_features(${PROJECT_NAME} PRIVATE cxx_std_20)
target_include_directories(${PROJECT_NAME} PRIVATE ${CMAKE_CURRENT_BINARY_DIR})
target_include_directories(${PROJECT_NAME} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
